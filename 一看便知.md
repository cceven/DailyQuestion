### Dijkstra算法

```cpp
vector<int> dijkstra(int n, vector<vector<pair<int, int>>> adj, int source) {
  // 初始化source到所有节点的距离都为无限大
  vector<int> dist(n + 1, INT_MAX);
  // souce到自己的距离为0
  dist[source] = 0;

  // 使用优先队列，每次取出最小距离的点
  priority_queue<
          pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>
          pq;
  // 首先放入source点
  pq.push({ 0, source });

  // 存储是否已经访问过
  vector<bool> is_visit(n + 1, false);
  // 开始算法
  while (!pq.empty()) {
    // 首先取出最小距离的点
    auto [d, u] = pq.top();
    pq.pop();
    // 如果已经访问过了，那么直接跳过
    if (is_visit[u]) continue;
    is_visit[u] = true;

    // 访问当前点可达的所有点，更新最小距离并push到优先队列中
    for (auto [v, weight] : adj[u]) {
      if (dist[u] + weight < dist[v]) {
        dist[v] = dist[u] + weight;
        pq.push({ dist[v], v });
      }
    }
  }
  // 返回source到所有点的最小距离
  return dist;
}
```

### 当mod为质数时，使用费马小定理

```cpp
// 快速幂算法
int fastPow(int a, int b, int mod) {
  a %= mod;
  int res = 1;
  while (b > 0) {
    if (b & 1) res = res * a % mod;
    a = a * a % mod;
    b >>= 1;
  }
  return res;
}

int getInv(int a, int mod) {
  return fastPow(a, mod - 2, mod);
}
```

### 一般情况时，使用扩展欧几里得定理

```cpp
void exgcd(int a, int b, int& x, int& y) {
  if (b == 0) {
    x = 1;
    y = 0;
  }

  exgcd(b, a % b, y, x);
  y -= (a / b) * x;
}

int getInv(int a, int mod) {
  int x = 1, y = 0;
  exgcd(a, mod, x, y);
  return (x % mod + mod) % mod;  // 防止出现负数
}
```

```cpp
void solve() {
  int n, m;
  cin >> n >> m;

  vector<vector<int>> adj(n + 1);
  vector<int> in_degree(n + 1, 0), out_degree(n + 1, 0);
  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    adj[u].push_back(v);
    out_degree[u]++;
    in_degree[v]++;
  }

  int start_node = -1, end_node = -1;
  for (int i = 1; i <= n; i++) {
    if (out_degree[i] - in_degree[i] == 1) {
      // 可能是开始节点，必须先判断开始节点再判断结束节点
      if (start_node != -1) {
        cout << "No" << endl;
        return;
      }
      start_node = i;
    } else if (in_degree[i] - out_degree[i] == 1) {
      // 可能是终点节点
      if (end_node != -1) {
        cout << "No" << endl;
        return;
      }

      end_node = i;
    } else if (out_degree[i] != in_degree[i]) {
      cout << "No" << endl;
      return;
    }

    sort(adj[i].begin(), adj[i].end(), greater<int>());
  }

  if (start_node == -1) start_node = 1;

  vector<pair<int, int>> ans;
  auto dfs = [&](auto &&self, int cur_node) -> void {
    while (adj[cur_node].size()) {
      int v = adj[cur_node].back();
      adj[cur_node].pop_back();
      self(self, v);
      ans.push_back({ cur_node, v });
    }
  };
  dfs(dfs, start_node);

  reverse(ans.begin(), ans.end());
  cout << ans[0].first << " ";
  for (int i = 0; i < ans.size(); i++) {
    cout << ans[i].second << " ";
  }
}

  cin.tie(nullptr);
  int t = 1;
  // cin >> t;
```

```cpp
#include <bits/stdc++.h>
#define endl "\n"
using namespace std;
using ll = long long;
#define int ll

int degress[1031]   = { 0 };  // 存储度
int adj[1031][1031] = { 0 };  // 邻接表

void solve() {
  int n          = 0, m;
  int start_node = INT_MAX;  // 最小的节点，也是开始节点
  cin >> m;

  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    n          = max({ n, u, v });  // 最大的节点
    start_node = min(u, v);         // 更新开始节点
    // 更新邻接表
    adj[u][v]++;
    adj[v][u]++;
    // 更新度
    degress[u]++;
    degress[v]++;
  }

  // 如果不是环形，重新找到起点
  for (int i = 1; i <= n; i++) {
    if (degress[i] & 1) {
      start_node = i;
      break;
    }
  }

  vector<int> ans;
  auto dfs = [&](auto &&self, int u) -> void {
    for (int i = 1; i <= n; i++) {
      while (adj[u][i] > 0) {
        adj[u][i]--;
        adj[i][u]--;
        self(self, i);
      }
    }

    ans.push_back(u);
  };

  dfs(dfs, start_node);
  reverse(ans.begin(), ans.end());
  for (int i = 0; i < ans.size(); i++) {
    cout << ans[i] << endl;
  }
}

signed main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t = 1;
  // cin >> t;
  while (t--) solve();
  return 0;
}
```

```cpp
struct UnionFind {
    vector<int> fa, sz;
    UnionFind(const int n): fa(n), sz(n, 1) { iota(fa.begin(), fa.end(), 0); }

    // 找到所属的类
    int findFa(const int x) {
      if (fa[x] != x) fa[x] = findFa(fa[x]);
      return fa[x];
    }

    // 合并类
    void unite(int from, int to) {
      int x = findFa(from);
      int y = findFa(to);
      fa[x] = y;
      sz[y] += sz[x];
    }

    // 判断是否属于同一类
    bool isSame(int x, int y) { return findFa(x) == findFa(y); }
```

### 通过中序遍历和后序遍历构造树

```cpp
TreeNode *buildTree(
        vector<int> in_order, int in_start, int in_end, vector<int> post_order,
        int post_start, int post_end, unordered_map<int, int> in_map
) {
  if (in_start > in_end || post_start > post_end) return nullptr;

  int root_val   = post_order[post_end];  // 找到根的值
  TreeNode *root = new TreeNode(root_val);

  int root_pos = in_map[root_val];     // 找到根在中序遍历中的位置
  int left_sz  = root_pos - in_start;  // 左子树的大小

  root->left = buildTree(
          in_order, in_start, root_pos - 1, post_order, post_start,
          post_start + left_sz - 1, in_map
  );
  root->right = buildTree(
          in_order, root_pos + 1, in_end, post_order, post_start + left_sz,
          post_end - 1, in_map
  );

  return root;
}
```

```cpp
struct Node {
    Node *son[26]{};
    bool is_end = false;  // 是否是一个单词的结尾
};

struct Trie {
    Node *root = new Node();

    // 查找单词
    int trieFind(string word) {
      Node *cur = root;
      for (auto c : word) {
        int j = c - 'a';
        if (cur->son[j] == nullptr) return 0;
        cur = cur->son[j];
      }
      return (cur->is_end ? 2 : 1);
    }

    // 插入单词
    void insert(string word) {
      Node *cur = root;
      for (auto &c : word) {
        int j = c - 'a';
        if (cur->son[j] == nullptr) cur->son[j] = new Node();
        cur = cur->son[j];
      }
      cur->is_end = true;
    }

    // 查找是否为一个单词
    bool search(string word) { return trieFind(word) == 2; };
    // 查找是否存在以prefix为前缀的单词
    bool startsWith(string prefix) { return trieFind(prefix) > 0; }
};
```
