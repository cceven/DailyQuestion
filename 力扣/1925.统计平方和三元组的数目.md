直接遍历。
```cpp
class Solution {
public:
    int countTriples(int n) {
        int ans = 0;
        for (int i = 3; i <= n - 2; ++i) {
            for (int j = i + 1; j <= n - 1; ++j) {
                int sum = i * i + j * j;
                if (sum > n*n)
                    break;//剪枝操作
                if (isSq(sum))
                    ans += 2;
            }
        }
        return ans;
    }

private:
    bool isSq(int num) {
        int n = sqrt(num);
        return n * n == num;
    }
};
```
