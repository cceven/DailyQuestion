[恋恋的01串大冒险](https://ac.nowcoder.com/acm/contest/124143/D)
使用一个数组`ans`来存储操作i次可以走到的最大位置。使用`zero_run`记录当前连续的`0`的个数，如果达到`k`此那么需要进行一次操作，并且更新`ans`数组并且重置`zero_run`。
![[Pasted image 20251214141028.png]]
```cpp
#include <bits/stdc++.h>
#define endl "\n"
using namespace std;
using ll = long long;
#define int ll

void solve() {
  int n, k;
  string s;
  cin >> n >> k >> s;

  vector<int> ans(n + 1, 0);  // 花费i个操作能走到第几步

  int zero_run = 0;  // 当前连续0的长度
  int cost     = 0;  // 当前操作数
  for (int i = 0; i < n; i++) {
    if (s[i] == '0')
      zero_run++;
    else
      zero_run = 0;

    // 需要操作一次
    if (zero_run == k) {
      ans[cost] = i;
      cost++;
      zero_run = 0;
    }
  }

  // 剩下的部分肯定能存在到最后
  for (int i = cost; i <= n; i++) {
    ans[i] = n;
  }
  // 输出答案
  for (int i = 0; i <= n; i++) {
    cout << ans[i] << " ";
  }
}

signed main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  int t = 1;
  // cin >> t;
  while (t--) solve();
  return 0;
}
```
